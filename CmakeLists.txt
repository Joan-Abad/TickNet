cmake_minimum_required(VERSION 3.20)

project(TickNet LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Warnings
if (MSVC)
    add_compile_options(/W4 /WX)
else()
    add_compile_options(-Wall -Wextra -Wpedantic -Werror)
endif()

add_library(TickNet STATIC)

target_include_directories(TickNet
    PUBLIC
        ${CMAKE_CURRENT_SOURCE_DIR}/include
)

file(GLOB_RECURSE TICKNET_SOURCES
    CONFIGURE_DEPENDS
    ${CMAKE_CURRENT_SOURCE_DIR}/src/*.cpp
    ${CMAKE_CURRENT_SOURCE_DIR}/include/*.h
)

target_sources(TickNet PRIVATE ${TICKNET_SOURCES})

#Sample Executables

add_executable(TickNetClient
    samples/Client/main.cpp
)

target_link_libraries(TickNetClient
    PRIVATE TickNet
)
